let now = new Date();

let now = new Date();

let minutes = now.getMinutes(); // Dakika alır (0-59 arası)
let seconds = now.getSeconds(); // Saniye alır (0-59 arası)
console.log(`Saat: ${hours}:${minutes}:${seconds}`);
